function rangeOfNumbers(startNum, endNum) {
 if (startNum >= endNum) {
  return [startNum];
 }
 else {
   const numArr = rangeOfNumbers(startNum, endNum -1);
   numArr.push(endNum);
   return numArr;
 }
}
